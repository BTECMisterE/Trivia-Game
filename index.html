// --- QR Setup with Force-Link --- // This ensures the student is sent to play.html with the ?room= parameter attached const currentURL = window.location.href; const baseURL = currentURL.substring(0, currentURL.lastIndexOf('/') + 1); const pUrl = baseURL + "play.html?room=" + ROOM; // Clear any old QR code before making a new one document.getElementById("qrcode").innerHTML = ""; new QRCode(document.getElementById("qrcode"), { text: pUrl, width: 100, height: 100, correctLevel : QRCode.CorrectLevel.H });
